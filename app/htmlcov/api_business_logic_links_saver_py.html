1'''  

2 Модуль содержит класс LinksSaver, который извлекает домены из ссылок 

3 и сохраняет в БД Redis.

5import re 

6from datetime import datetime 

7 

8 

9class LinksSaver: 

10 ''' 

11 Пример использования: 

12 links_saver = LinksSaver(data["links"], datetime.datetime.now(tz=tz), redis_instance) 

13 response = links_saver.save() 

14  

15 Возвращает словарь {"status": "ok"}, если нет ошибок.  

16 Если ошибки есть, {"status": "warning", description: "описание"}  

17 или {"status": "error", description: "описание"}  

18 ''' 

19 

20 def __init__(self, links_list, timestamp: datetime, redis_instance): 

21 self.links_list = links_list 

22 self.timestamp = timestamp 

23 self.redis_instance = redis_instance 

24 self.response = {"status": "ok"} 

25 

26 def save(self): 

27 self.exctract_domains() 

28 self.remove_dublicates() 

29 self.save_to_redis() 

30 return self.response 

31 

32 def exctract_domains(self): 

33 pattern = r"[a-z0-9\-_]{2,}\.[a-z]{2,4}" 

34 for index, link in enumerate(self.links_list): 

35 try: 

36 self.links_list[index] = re.search(pattern, link).group(0) 

37 except AttributeError: 

38 self.add_warning_about_bad_link(link) 

39 

40 def add_warning_about_bad_link(self, link): 

41 msg = f"Warning! Link '{link}' has bad format and didn`t save to database! " 

42 self.response["status"] = "warning" 

43 if self.response.get("description"): 

44 self.response["description"] += msg 

45 else: 

46 self.response["description"] = msg 

47 

48 def remove_dublicates(self): 

49 self.links_list = list(set(self.links_list)) 

50 

51 def save_to_redis(self): 

52 key = self.timestamp.isoformat().split('.')[0] 

53 try: 

54 self.redis_instance.lpush(key, *self.links_list) 

55 except Exception as e: 

56 self.response["status"] = "error" 

57 self.response["description"] = f"{e}"
